66  void _lazy_init()
67  {
68  // this gets called before static constructors
69  // so make sure we only get called once.
70  static bool once = true;
71  if (once)
72  {
73  once = false;
74 
75  // call our own constructor. Global new can be called before the constructors are called.
76  // Make sure it gets called by having it done lazily in the call to allocate.
77  new (this) AllocatorStrategyGlobalSmallBlock();
78  }

81  AllocatorStrategyGlobalSmallBlock()
82  {
83  // this gets called before static constructors
84  // so make sure we only get called once.
85  static bool once = true;
86  if (once)
87  {
88  once = false;
89 
90  _maxBlockSize = 1 << kMaxSmallBlockPower;
91 
92 #if CC_ENABLE_ALLOCATOR_DIAGNOSTICS
93  AllocatorDiagnostics::instance()->trackAllocator(this);
94  AllocatorBase::setTag("GlobalSmallBlock");
95 #endif
96 
97  memset(_smallBlockAllocators, 0, sizeof(_smallBlockAllocators));
98 #if CC_ENABLE_ALLOCATOR_DIAGNOSTICS
99  memset(_smallBlockAllocations, 0, sizeof(_smallBlockAllocations));
100 #endif
101  // cannot call new on the allocator here because it will recurse
102  // so instead we allocate from the global allocator and construct in place.
103  #define SBA(n, size) \
104  if (size <= _maxBlockSize) \
105  { \
106  auto v = ccAllocatorGlobal.allocate(sizeof(SType(size))); \
107  _smallBlockAllocators[n] = (AllocatorBase*)(new (v) SType(size)("GlobalSmallBlock::"#size)); \
108  }

110  SBA(2, 4)
111  SBA(3, 8);
112  SBA(4, 16);
113  SBA(5, 32);
114  SBA(6, 64);

116  SBA(8, 256);
117  SBA(9, 512);

123  #undef SBA

138  // @brief Allocate a block of some size. If the block is <= 8192 it is allocated out of an array
139  // of fixed size block allocators. If larger, then we default back to the global allocator.
140  // @param size Size of block to allocate. This will be rounded to the next power of two.
141  CC_ALLOCATOR_INLINE void* allocate(size_t size)
142  {
143  _lazy_init();
144 
145  if (size < sizeof(intptr_t)) // always allocate at least enough space to store a pointer. this is
146  size = sizeof(intptr_t); // so we can link the empty blocks together in the block allocator.
147 
148  // if the size is greater than what we determine to be a small block
149  // size then fall through to calling the global allocator instead.
150  if (size > _maxBlockSize)
151  return ccAllocatorGlobal.allocate(size);
152 
153  // make sure the size fits into one of the
154  // fixed sized block allocators we have above.
155  size_t adjusted_size = AllocatorBase::nextPow2BlockSize(size);
156 
157  #define ALLOCATE(slot, size) \
158  case size: \
159  { \
160  void* v = _smallBlockAllocators[slot]; \
161  CC_ASSERT(nullptr != v); \
162  auto a = (SType(size)*)v; \
163  address = a->allocate(adjusted_size); \
164  TRACK(slot, size, +=); \
165  } \
166  break;
167 
168  void* address = nullptr;
169 
170  switch (adjusted_size)
171  {
172  ALLOCATE(2, 4);
173  ALLOCATE(3, 8);
174  ALLOCATE(4, 16);
175  ALLOCATE(5, 32);
176  ALLOCATE(6, 64);
177  ALLOCATE(7, 128);
178  ALLOCATE(8, 256);
179  ALLOCATE(9, 512);
180  ALLOCATE(10, 1024);
181  ALLOCATE(11, 2048);
182  ALLOCATE(12, 4096);
183  ALLOCATE(13, 8192);
184  default:
185  CC_ASSERT(false);
186  break;
187  }
188 
189  #undef ALLOCATE
190 
191  CC_ASSERT(adjusted_size < AllocatorBase::kDefaultAlignment || 0 == ((intptr_t)address & (AllocatorBase::kDefaultAlignment - 1)));
192  CC_ASSERT(nullptr != address);
193 
194  return address;
195  }

197  // @brief Deallocate a block by choosing one of the fixed size block allocators
198  // or defaulting to the global allocator if we do not own this block.
199  CC_ALLOCATOR_INLINE void deallocate(void* address, size_t size = 0)
200  {
201  // if we didn't get a size, then we need to find the allocator
202  // by asking each if they own the block. For allocators that
203  // have few large pages, this is extremely fast.
204  if (0 == size)
205  {
206  #define OWNS(slot, S, address) \
207  case S: \
208  { \
209  void* v = _smallBlockAllocators[slot]; \
210  if (v) \
211  { \
212  auto a = (SType(S)*)v; \
213  if (a->owns(address)) \
214  { \
215  size = SType(S)::block_size; \
216  break; \
217  } \
218  } \
219  }
220 
221  // falls through until found
222  switch (sizeof(uint32_t))
223  {
224  OWNS(2, 4, address);
225  OWNS(3, 8, address);
226  OWNS(4, 16, address);
227  OWNS(5, 32, address);
228  OWNS(6, 64, address);
229  OWNS(7, 128, address);
230  OWNS(8, 256, address);
231  OWNS(9, 512, address);
232  OWNS(10, 1024, address);
233  OWNS(11, 2048, address);
234  OWNS(12, 4096, address);
235  OWNS(13, 8192, address);
236  }
237  }
238 
239  // if the size is greater than what we determine to be a small block
240  // size then default to calling the global allocator instead.
241  if (0 == size || size > _maxBlockSize)
242  return ccAllocatorGlobal.deallocate(address, size);
243 
244  if (size < sizeof(intptr_t)) // always allocate at least enough space to store a pointer. this is
245  size = sizeof(intptr_t); // so we can link the empty blocks together in the block allocator.
246 
247  // make sure the size fits into one of the
248  // fixed sized block allocators we have above.
249  size_t adjusted_size = AllocatorBase::nextPow2BlockSize(size);
250 
251  #define DEALLOCATE(slot, size, address) \
252  case size: \
253  { \
254  void* v = _smallBlockAllocators[slot]; \
255  CC_ASSERT(nullptr != v); \
256  auto a = (SType(size)*)v; \
257  a->deallocate(address, size); \
258  TRACK(slot, size, -=); \
259  } \
260  break;
261 
262  switch (adjusted_size)
263  {
264  DEALLOCATE(2, 4, address);
265  DEALLOCATE(3, 8, address);
266  DEALLOCATE(4, 16, address);
267  DEALLOCATE(5, 32, address);
268  DEALLOCATE(6, 64, address);
269  DEALLOCATE(7, 128, address);
270  DEALLOCATE(8, 256, address);
271  DEALLOCATE(9, 512, address);
272  DEALLOCATE(10, 1024, address);
273  DEALLOCATE(11, 2048, address);
274  DEALLOCATE(12, 4096, address);
275  DEALLOCATE(13, 8192, address);
276  default:
277  CC_ASSERT(false);
278  }
279 
280  #undef DEALLOCATE
281  }
