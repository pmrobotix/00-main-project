1483 /*
1484  * IMPORTANT NOTICE!
1485  *
1486  * When sending with websocket protocol
1487  *
1488  * LWS_WRITE_TEXT,
1489  * LWS_WRITE_BINARY,
1490  * LWS_WRITE_CONTINUATION,
1491  * LWS_WRITE_PING,
1492  * LWS_WRITE_PONG
1493  *
1494  * the send buffer has to have LWS_PRE bytes valid BEFORE
1495  * the buffer pointer you pass to lws_write().
1496  *
1497  * This allows us to add protocol info before and after the data, and send as
1498  * one packet on the network without payload copying, for maximum efficiency.
1499  *
1500  * So for example you need this kind of code to use lws_write with a
1501  * 128-byte payload
1502  *
1503  * char buf[LWS_PRE + 128];
1504  *
1505  * // fill your part of the buffer... for example here it's all zeros
1506  * memset(&buf[LWS_PRE], 0, 128);
1507  *
1508  * lws_write(wsi, &buf[LWS_PRE], 128, LWS_WRITE_TEXT);
1509  *
1510  * When sending HTTP, with
1511  *
1512  * LWS_WRITE_HTTP,
1513  * LWS_WRITE_HTTP_HEADERS
1514  * LWS_WRITE_HTTP_FINAL
1515  *
1516  * there is no protocol data prepended, and don't need to take care about the
1517  * LWS_PRE bytes valid before the buffer pointer.
1518  *
1519  * LWS_PRE is at least the frame nonce + 2 header + 8 length
1520  * LWS_SEND_BUFFER_POST_PADDING is deprecated, it's now 0 and can be left off.
1521  * The example apps no longer use it.
1522  *
1523  * Pad LWS_PRE to the CPU word size, so that word references
1524  * to the address immediately after the padding won't cause an unaligned access
1525  * error. Sometimes for performance reasons the recommended padding is even
1526  * larger than sizeof(void *).
1527  */
