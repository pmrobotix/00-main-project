57 template<typename CharT, typename Traits, typename Allocator>
58 class ResizableBufferAdapter< std::basic_string<CharT, Traits, Allocator> > : public ResizableBuffer {
59  typedef std::basic_string<CharT, Traits, Allocator> BufferType;
60  BufferType* _buffer;
61 public:
62  explicit ResizableBufferAdapter(BufferType* buffer) : _buffer(buffer) {}
63  virtual void resize(size_t size) override {
64  _buffer->resize((size + sizeof(CharT) - 1) / sizeof(CharT));
65  }
66  virtual void* buffer() const override {
67  // can not invoke string::front() if it is empty
68 
69  if (_buffer->empty())
70  return nullptr;
71  else
72  return &_buffer->front();
73  }

96 template<>
97 class ResizableBufferAdapter<Data> : public ResizableBuffer {
98  typedef Data BufferType;
99  BufferType* _buffer;
100 public:
101  explicit ResizableBufferAdapter(BufferType* buffer) : _buffer(buffer) {}
102  virtual void resize(size_t size) override {
103  if (static_cast<size_t>(_buffer->getSize()) < size) {
104  auto old = _buffer->getBytes();
105  void* buffer = realloc(old, size);
106  if (buffer)
107  _buffer->fastSet((unsigned char*)buffer, size);
108  }
109  }
110  virtual void* buffer() const override {
111  return _buffer->getBytes();
112  }
