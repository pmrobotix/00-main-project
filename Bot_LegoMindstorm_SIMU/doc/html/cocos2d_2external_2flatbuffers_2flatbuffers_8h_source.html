95 template<typename T> T EndianScalar(T t) {
96  #if FLATBUFFERS_LITTLEENDIAN
97  return t;
98  #else
99  #if defined(_MSC_VER)
100  #pragma push_macro("__builtin_bswap16")
101  #pragma push_macro("__builtin_bswap32")
102  #pragma push_macro("__builtin_bswap64")
103  #define __builtin_bswap16 _byteswap_ushort
104  #define __builtin_bswap32 _byteswap_ulong
105  #define __builtin_bswap64 _byteswap_uint64
106  #endif
107  // If you're on the few remaining big endian platforms, we make the bold
108  // assumption you're also on gcc/clang, and thus have bswap intrinsics:
109  if (sizeof(T) == 1) { // Compile-time if-then's.
110  return t;
111  } else if (sizeof(T) == 2) {
112  auto r = __builtin_bswap16(*reinterpret_cast<uint16_t *>(&t));
113  return *reinterpret_cast<T *>(&r);
114  } else if (sizeof(T) == 4) {
115  auto r = __builtin_bswap32(*reinterpret_cast<uint32_t *>(&t));
116  return *reinterpret_cast<T *>(&r);
117  } else if (sizeof(T) == 8) {
118  auto r = __builtin_bswap64(*reinterpret_cast<uint64_t *>(&t));
119  return *reinterpret_cast<T *>(&r);
120  } else {
121  assert(0);
122  }
123  #if defined(_MSC_VER)
124  #pragma pop_macro("__builtin_bswap16")
125  #pragma pop_macro("__builtin_bswap32")
126  #pragma pop_macro("__builtin_bswap64")
127  #endif
128  #endif
129 }

147 // When we read serialized data from memory, in the case of most scalars,
148 // we want to just read T, but in the case of Offset, we want to actually
149 // perform the indirection and return a pointer.
150 // The template specialization below does just that.
151 // It is wrapped in a struct since function templates can't overload on the
152 // return type like this.
153 // The typedef is for the convenience of callers of this function
154 // (avoiding the need for a trailing return decltype)
155 template<typename T> struct IndirectHelper {
156  typedef T return_type;
157  static const size_t element_stride = sizeof(T);
158  static return_type Read(const uint8_t *p, uoffset_t i) {
159  return EndianScalar((reinterpret_cast<const T *>(p))[i]);
160  }
161 };
162 template<typename T> struct IndirectHelper<Offset<T>> {
163  typedef const T *return_type;
164  static const size_t element_stride = sizeof(uoffset_t);
165  static return_type Read(const uint8_t *p, uoffset_t i) {
166  p += i * sizeof(uoffset_t);
167  return reinterpret_cast<return_type>(p + ReadScalar<uoffset_t>(p));
168  }
169 };
170 template<typename T> struct IndirectHelper<const T *> {
171  typedef const T *return_type;
172  static const size_t element_stride = sizeof(T);
173  static return_type Read(const uint8_t *p, uoffset_t i) {
174  return reinterpret_cast<const T *>(p + i * sizeof(T));

304 // This is a minimal replication of std::vector<uint8_t> functionality,
305 // except growing from higher to lower addresses. i.e push_back() inserts data
306 // in the lowest address in the vector.
307 class vector_downward {
308  public:
309  explicit vector_downward(size_t initial_size,
310  const simple_allocator &allocator)
311  : reserved_(initial_size),
312  buf_(allocator.allocate(reserved_)),
313  cur_(buf_ + reserved_),
314  allocator_(allocator) {
315  assert((initial_size & (sizeof(largest_scalar_t) - 1)) == 0);
316  }
317 
318  ~vector_downward() { allocator_.deallocate(buf_); }
319 
320  void clear() { cur_ = buf_ + reserved_; }
321 
322  size_t growth_policy(size_t bytes) {
323  return (bytes / 2) & ~(sizeof(largest_scalar_t) - 1);
324  }
325 
326  uint8_t *make_space(size_t len) {
327  if (buf_ > cur_ - len) {
328  auto old_size = size();
329  reserved_ += std::max(len, growth_policy(reserved_));
330  auto new_buf = allocator_.allocate(reserved_);
331  auto new_cur = new_buf + reserved_ - old_size;
332  memcpy(new_cur, cur_, old_size);
333  cur_ = new_cur;
334  allocator_.deallocate(buf_);
335  buf_ = new_buf;
336  }
337  cur_ -= len;
338  // Beyond this, signed offsets may not have enough range:
339  // (FlatBuffers > 2GB not supported).
340  assert(size() < (1UL << (sizeof(soffset_t) * 8 - 1)) - 1);
341  return cur_;
342  }
343 
344  uoffset_t size() const {
345  return static_cast<uoffset_t>(reserved_ - (cur_ - buf_));
346  }
347 
348  uint8_t *data() const { return cur_; }
349 
350  uint8_t *data_at(size_t offset) { return buf_ + reserved_ - offset; }
351 
352  // push() & fill() are most frequently called with small byte counts (<= 4),
353  // which is why we're using loops rather than calling memcpy/memset.
354  void push(const uint8_t *bytes, size_t num) {
355  auto dest = make_space(num);
356  for (size_t i = 0; i < num; i++) dest[i] = bytes[i];
357  }
358 
359  void fill(size_t zero_pad_bytes) {
360  auto dest = make_space(zero_pad_bytes);
361  for (size_t i = 0; i < zero_pad_bytes; i++) dest[i] = 0;
362  }
363 
364  void pop(size_t bytes_to_remove) { cur_ += bytes_to_remove; }
365 
366  private:
367  // You shouldn't really be copying instances of this class.
368  vector_downward(const vector_downward &);
369  vector_downward &operator=(const vector_downward &);
370 
371  size_t reserved_;
372  uint8_t *buf_;
373  uint8_t *cur_; // Points at location between empty (below) and used (above).
374  const simple_allocator &allocator_;
375 };

512  // This finishes one serialized object by generating the vtable if it's a
513  // table, comparing it against existing vtables, and writing the
514  // resulting vtable offset.
515  uoffset_t EndTable(uoffset_t start, voffset_t numfields) {
516  // Write the vtable offset, which is the start of any Table.
517  // We fill it's value later.
518  auto vtableoffsetloc = PushElement<uoffset_t>(0);
519  // Write a vtable, which consists entirely of voffset_t elements.
520  // It starts with the number of offsets, followed by a type id, followed
521  // by the offsets themselves. In reverse:
522  buf_.fill(numfields * sizeof(voffset_t));
523  auto table_object_size = vtableoffsetloc - start;
524  assert(table_object_size < 0x10000); // Vtable use 16bit offsets.
525  PushElement<voffset_t>(static_cast<voffset_t>(table_object_size));
526  PushElement<voffset_t>(FieldIndexToOffset(numfields));
527  // Write the offsets into the table
528  for (auto field_location = offsetbuf_.begin();
529  field_location != offsetbuf_.end();
530  ++field_location) {
531  auto pos = static_cast<voffset_t>(vtableoffsetloc - field_location->off);
532  // If this asserts, it means you've set a field twice.
533  assert(!ReadScalar<voffset_t>(buf_.data() + field_location->id));
534  WriteScalar<voffset_t>(buf_.data() + field_location->id, pos);
535  }
536  offsetbuf_.clear();
537  auto vt1 = reinterpret_cast<voffset_t *>(buf_.data());
538  auto vt1_size = ReadScalar<voffset_t>(vt1);
539  auto vt_use = GetSize();
540  // See if we already have generated a vtable with this exact same
541  // layout before. If so, make it point to the old one, remove this one.
542  for (auto it = vtables_.begin(); it != vtables_.end(); ++it) {
543  if (memcmp(buf_.data_at(*it), vt1, vt1_size)) continue;
544  vt_use = *it;
545  buf_.pop(GetSize() - vtableoffsetloc);
546  break;
547  }
548  // If this is a new vtable, remember it.
549  if (vt_use == GetSize()) {
550  vtables_.push_back(vt_use);
551  }
552  // Fill the vtable offset we created above.
553  // The offset points from the beginning of the object to where the
554  // vtable is stored.
555  // Offsets default direction is downward in memory for future format
556  // flexibility (storing all vtables at the start of the file).
557  WriteScalar(buf_.data_at(vtableoffsetloc),
558  static_cast<soffset_t>(vt_use) -
559  static_cast<soffset_t>(vtableoffsetloc));
560  return vtableoffsetloc;
561  }

853 // "tables" use an offset table (possibly shared) that allows fields to be
854 // omitted and added at will, but uses an extra indirection to read.

856  public:
857  // This gets the field offset for any of the functions below it, or 0
858  // if the field was not present.
859  voffset_t GetOptionalFieldOffset(voffset_t field) const {
860  // The vtable offset is always at the start.
861  auto vtable = data_ - ReadScalar<soffset_t>(data_);
862  // The first element is the size of the vtable (fields + type id + itself).
863  auto vtsize = ReadScalar<voffset_t>(vtable);

865  // data, so it's the same as if the offset was 0 (not present).
866  return field < vtsize ? ReadScalar<voffset_t>(vtable + field) : 0;
