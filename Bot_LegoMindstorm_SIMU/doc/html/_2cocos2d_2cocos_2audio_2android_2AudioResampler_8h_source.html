69  // Resample int16_t samples from provider and accumulate into 'out'.
70  // A mono provider delivers a sequence of samples.
71  // A stereo provider delivers a sequence of interleaved pairs of samples.
72  //
73  // In either case, 'out' holds interleaved pairs of fixed-point Q4.27.
74  // That is, for a mono provider, there is an implicit up-channeling.
75  // Since this method accumulates, the caller is responsible for clearing 'out' initially.
76  //
77  // For a float resampler, 'out' holds interleaved pairs of float samples.
78  //
79  // Multichannel interleaved frames for n > 2 is supported for quality DYN_LOW_QUALITY,
80  // DYN_MED_QUALITY, and DYN_HIGH_QUALITY.
81  //
82  // Returns the number of frames resampled into the out buffer.
83  virtual size_t resample(int32_t* out, size_t outFrameCount,
84  AudioBufferProvider* provider) = 0;

125  // returns the inFrameCount required to generate outFrameCount frames.
126  //
127  // Placed here to be a consistent for all resamplers.
128  //
129  // Right now, we use the upper bound without regards to the current state of the
130  // input buffer using integer arithmetic, as follows:
131  //
132  // (static_cast<uint64_t>(outFrameCount)*mInSampleRate + (mSampleRate - 1))/mSampleRate;
133  //
134  // The double precision equivalent (float may not be precise enough):
135  // ceil(static_cast<double>(outFrameCount) * mInSampleRate / mSampleRate);
136  //
137  // this relies on the fact that the mPhaseIncrement is rounded down from
138  // #phases * mInSampleRate/mSampleRate and the fact that Sum(Floor(x)) <= Floor(Sum(x)).
139  // http://www.proofwiki.org/wiki/Sum_of_Floors_Not_Greater_Than_Floor_of_Sums
140  //
141  // (so long as double precision is computed accurately enough to be considered
142  // greater than or equal to the Floor(x) value in int32_t arithmetic; thus this
143  // will not necessarily hold for floats).
144  //
145  // TODO:
146  // Greater accuracy and a tight bound is obtained by:
147  // 1) subtract and adjust for the current state of the AudioBufferProvider buffer.
148  // 2) using the exact integer formula where (ignoring 64b casting)
149  // inFrameCount = (mPhaseIncrement * (outFrameCount - 1) + mPhaseFraction) / phaseWrapLimit;
150  // phaseWrapLimit is the wraparound (1 << kNumPhaseBits), if not specified explicitly.
151  //
152  inline size_t getInFrameCountRequired(size_t outFrameCount) {
153  return (static_cast<uint64_t>(outFrameCount)*mInSampleRate
154  + (mSampleRate - 1))/mSampleRate;
155  }
