43 // Convert an integer or floating point value to a string.
44 // In contrast to std::stringstream, "char" values are
45 // converted to a string of digits.
46 template<typename T> std::string NumToString(T t) {
47  // to_string() prints different numbers of digits for floats depending on
48  // platform and isn't available on Android, so we use stringstream
49  std::stringstream ss;
50  ss << t;
51  return ss.str();
52 }

54 template<> inline std::string NumToString<signed char>(signed char t) {
55  return NumToString(static_cast<int>(t));
56 }

58  return NumToString(static_cast<int>(t));

84 // false otherwise. If "binary" is false data is read

88  std::ifstream ifs(name, binary ? std::ifstream::binary : std::ifstream::in);

189 // Convert a unicode code point into a UTF-8 representation by appending it
190 // to a string. Returns the number of bytes generated.
191 inline int ToUTF8(uint32_t ucc, std::string *out) {
192  assert(!(ucc & 0x80000000)); // Top bit can't be set.
193  // 6 possible encodings: http://en.wikipedia.org/wiki/UTF-8
194  for (int i = 0; i < 6; i++) {
195  // Max bits this encoding can represent.
196  uint32_t max_bits = 6 + i * 5 + static_cast<int>(!i);
197  if (ucc < (1u << max_bits)) { // does it fit?
198  // Remaining bits not encoded in the first byte, store 6 bits each
199  uint32_t remain_bits = i * 6;
200  // Store first byte:
201  (*out) += static_cast<char>((0xFE << (max_bits - remain_bits)) |
202  (ucc >> remain_bits));
203  // Store remaining bytes:
204  for (int j = i - 1; j >= 0; j--) {
205  (*out) += static_cast<char>(((ucc >> (j * 6)) & 0x3F) | 0x80);
206  }
207  return i + 1; // Return the number of bytes added.
208  }
209  }
210  assert(0); // Impossible to arrive here.
211  return -1;
212 }

214 // Converts whatever prefix of the incoming string corresponds to a valid
215 // UTF-8 sequence into a unicode code. The incoming pointer will have been
216 // advanced past all bytes parsed.
217 // returns -1 upon corrupt UTF-8 encoding (ignore the incoming pointer in
218 // this case).
219 inline int FromUTF8(const char **in) {
220  int len = 0;
221  // Count leading 1 bits.
222  for (int mask = 0x80; mask >= 0x04; mask >>= 1) {
223  if (**in & mask) {
224  len++;
225  } else {
226  break;
227  }
228  }
229  if ((**in << len) & 0x80) return -1; // Bit after leading 1's must be 0.
230  if (!len) return *(*in)++;
231  // Grab initial bits of the code.
232  int ucc = *(*in)++ & ((1 << (7 - len)) - 1);
233  for (int i = 0; i < len - 1; i++) {
234  if ((**in & 0xC0) != 0x80) return -1; // Upper bits must 1 0.
235  ucc <<= 6;
236  ucc |= *(*in)++ & 0x3F; // Grab 6 more bits of the code.
237  }
238  return ucc;
239 }
