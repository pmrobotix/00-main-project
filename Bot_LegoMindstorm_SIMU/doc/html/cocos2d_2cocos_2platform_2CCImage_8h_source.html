37 // premultiply alpha, or the effect will wrong when want to use other pixel format in Texture2D,
38 // such as RGB888, RGB5A1
39 #define CC_RGB_PREMULTIPLY_ALPHA(vr, vg, vb, va) \
40  (unsigned)(((unsigned)((unsigned char)(vr) * ((unsigned char)(va) + 1)) >> 8) | \
41  ((unsigned)((unsigned char)(vg) * ((unsigned char)(va) + 1) >> 8) << 8) | \
42  ((unsigned)((unsigned char)(vb) * ((unsigned char)(va) + 1) >> 8) << 16) | \
43  ((unsigned)(unsigned char)(va) << 24))
