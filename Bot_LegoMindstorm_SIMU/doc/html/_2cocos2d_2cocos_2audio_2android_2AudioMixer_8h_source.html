86  // Only creates a sample rate converter the first time that
87  // the track sample rate is different from the mix sample rate.
88  // If the new sample rate is the same as the mix sample rate,
89  // and a sample rate converter already exists,
90  // then the sample rate converter remains present but is a no-op.
91  RESET = 0x4101, // Reset sample rate converter without changing sample rate.
92  // This clears out the resampler's input buffer.
93  REMOVE = 0x4102, // Remove the sample rate converter on this track name;
94  // the track is restored to the mix sample rate.
95  // for target RAMP_VOLUME and VOLUME (8 channels max)
96  // FIXME use float for these 3 to improve the dynamic range

98  VOLUME1 = 0x4201,
