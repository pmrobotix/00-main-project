62 template<typename CharT, typename Traits, typename Allocator>
63 class ResizableBufferAdapter< std::basic_string<CharT, Traits, Allocator> > : public ResizableBuffer {
64  typedef std::basic_string<CharT, Traits, Allocator> BufferType;
65  BufferType* _buffer;
66 public:
67  explicit ResizableBufferAdapter(BufferType* buffer) : _buffer(buffer) {}
68  virtual void resize(size_t size) override {
69  _buffer->resize((size + sizeof(CharT) - 1) / sizeof(CharT));
70  }
71  virtual void* buffer() const override {
72  // can not invoke string::front() if it is empty
73 
74  if (_buffer->empty())

101 template<>
102 class ResizableBufferAdapter<Data> : public ResizableBuffer {
103  typedef Data BufferType;
104  BufferType* _buffer;
105 public:
106  explicit ResizableBufferAdapter(BufferType* buffer) : _buffer(buffer) {}
107  virtual void resize(size_t size) override {
108  size_t oldSize = static_cast<size_t>(_buffer->getSize());
109  if (oldSize != size) {
110  auto old = _buffer->getBytes();
111  void* buffer = realloc(old, size);
112  if (buffer)
113  _buffer->fastSet((unsigned char*)buffer, size);
114  }
115  }
116  virtual void* buffer() const override {
117  return _buffer->getBytes();
118  }

987  template<typename T, typename R, typename ...ARGS>
988  static void performOperationOffthread(T&& action, R&& callback, ARGS&& ...args)
989  {
990 
991  // Visual Studio 2013 does not support using std::bind to forward template parameters into
992  // a lambda. To get around this, we will just copy these arguments via lambda capture
993 #if defined(_MSC_VER) && _MSC_VER < 1900
994  auto lambda = [action, callback, args...]()
995  {
996  Director::getInstance()->getScheduler()->performFunctionInCocosThread(std::bind(callback, action(args...)));
997  };
998 #else
999  // As cocos2d-x uses c++11, we will use std::bind to leverage move sematics to
1000  // move our arguments into our lambda, to potentially avoid copying.
1001  auto lambda = std::bind([](const T& actionIn, const R& callbackIn, const ARGS& ...argsIn)
1002  {
1003  Director::getInstance()->getScheduler()->performFunctionInCocosThread(std::bind(callbackIn, actionIn(argsIn...)));
1004  }, std::forward<T>(action), std::forward<R>(callback), std::forward<ARGS>(args)...);
1005 
1006 #endif
1007 
1008  AsyncTaskPool::getInstance()->enqueue(AsyncTaskPool::TaskType::TASK_IO, [](void*){}, nullptr, std::move(lambda));
1009  }
