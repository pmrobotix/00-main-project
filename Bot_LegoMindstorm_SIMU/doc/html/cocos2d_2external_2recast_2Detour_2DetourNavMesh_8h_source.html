512  inline dtPolyRef encodePolyId(unsigned int salt, unsigned int it, unsigned int ip) const
513  {
514 #ifdef DT_POLYREF64
515  return ((dtPolyRef)salt << (DT_POLY_BITS+DT_TILE_BITS)) | ((dtPolyRef)it << DT_POLY_BITS) | (dtPolyRef)ip;
516 #else
517  return ((dtPolyRef)salt << (m_polyBits+m_tileBits)) | ((dtPolyRef)it << m_polyBits) | (dtPolyRef)ip;
518 #endif
519  }
520 
528  inline void decodePolyId(dtPolyRef ref, unsigned int& salt, unsigned int& it, unsigned int& ip) const
529  {
530 #ifdef DT_POLYREF64
531  const dtPolyRef saltMask = ((dtPolyRef)1<<DT_SALT_BITS)-1;
532  const dtPolyRef tileMask = ((dtPolyRef)1<<DT_TILE_BITS)-1;
533  const dtPolyRef polyMask = ((dtPolyRef)1<<DT_POLY_BITS)-1;
534  salt = (unsigned int)((ref >> (DT_POLY_BITS+DT_TILE_BITS)) & saltMask);
535  it = (unsigned int)((ref >> DT_POLY_BITS) & tileMask);
536  ip = (unsigned int)(ref & polyMask);
537 #else
538  const dtPolyRef saltMask = ((dtPolyRef)1<<m_saltBits)-1;
539  const dtPolyRef tileMask = ((dtPolyRef)1<<m_tileBits)-1;
540  const dtPolyRef polyMask = ((dtPolyRef)1<<m_polyBits)-1;
541  salt = (unsigned int)((ref >> (m_polyBits+m_tileBits)) & saltMask);
542  it = (unsigned int)((ref >> m_polyBits) & tileMask);
543  ip = (unsigned int)(ref & polyMask);
544 #endif
545  }
