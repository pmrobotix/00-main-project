82  _locked.push_back(*_releasedIter);
83  _releasedIter = _released.erase(_releasedIter);

92  T *ptr = getFirst();
93  while (ptr)

95  if (ptr == data) break;

112  if (_releasedIter == _released.end()) return nullptr;
