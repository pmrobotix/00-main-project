185 /*
186  A dynamic array of Plain Old Data. Doesn't support constructors, etc.
187  Has a small initial memory pool, so that low or no usage will not
188  cause a call to new/delete
189 */
190 template <class T, int INIT>
191 class CC_DLL DynArray
192 {
193 public:
194  DynArray< T, INIT >() {
195  _mem = _pool;
196  _allocated = INIT;
197  _size = 0;
198  }
199 
200  ~DynArray() {
201  if ( _mem != _pool ) {
202  delete [] _mem;
203  }
204  }
205 
206  void Push( T t ) {
207  EnsureCapacity( _size+1 );
208  _mem[_size++] = t;
209  }
210 
211  T* PushArr( int count ) {
212  EnsureCapacity( _size+count );
213  T* ret = &_mem[_size];
214  _size += count;
215  return ret;
216  }
217 
218  T Pop() {
219  return _mem[--_size];
220  }
221 
222  void PopArr( int count ) {
223  TIXMLASSERT( _size >= count );
224  _size -= count;
225  }
226 
227  bool Empty() const {
228  return _size == 0;
229  }
230 
231  T& operator[](int i) {
232  TIXMLASSERT( i>= 0 && i < _size );
233  return _mem[i];
234  }
235 
236  const T& operator[](int i) const {
237  TIXMLASSERT( i>= 0 && i < _size );
238  return _mem[i];
239  }
240 
241  int Size() const {
242  return _size;
243  }
244 
245  int Capacity() const {
246  return _allocated;
247  }
248 
249  const T* Mem() const {
250  return _mem;
251  }
252 
253  T* Mem() {
254  return _mem;
255  }
256 
257 private:
258  void EnsureCapacity( int cap ) {
259  if ( cap > _allocated ) {
260  int newAllocated = cap * 2;
261  T* newMem = new T[newAllocated];
262  memcpy( newMem, _mem, sizeof(T)*_size ); // warning: not using constructors, only works for PODs
263  if ( _mem != _pool ) {
264  delete [] _mem;
265  }
266  _mem = newMem;
267  _allocated = newAllocated;
268  }
269  }
270 
271  T* _mem;
272  T _pool[INIT];
273  int _allocated; // objects allocated
274  int _size; // number objects in use
275 };

295 /*
296  Template child class to create pools of the correct type.
297 */
298 template< int SIZE >
299 class CC_DLL MemPoolT : public MemPool
300 {
301 public:
302  MemPoolT() : _root(0), _currentAllocs(0), _nAllocs(0), _maxAllocs(0), _nUntracked(0) {}
303  ~MemPoolT() {
304  // Delete the blocks.
305  for( int i=0; i<_blockPtrs.Size(); ++i ) {
306  delete _blockPtrs[i];
307  }
308  }
309 
310  virtual int ItemSize() const {
311  return SIZE;
312  }
313  int CurrentAllocs() const {
314  return _currentAllocs;
315  }
316 
317  virtual void* Alloc() {
318  if ( !_root ) {
319  // Need a new block.
320  Block* block = new Block();
321  _blockPtrs.Push( block );
322 
323  for( int i=0; i<COUNT-1; ++i ) {
324  block->chunk[i].next = &block->chunk[i+1];
325  }
326  block->chunk[COUNT-1].next = 0;
327  _root = block->chunk;
328  }
329  void* result = _root;
330  _root = _root->next;
331 
332  ++_currentAllocs;
333  if ( _currentAllocs > _maxAllocs ) {
334  _maxAllocs = _currentAllocs;
335  }
336  _nAllocs++;
337  _nUntracked++;
338  return result;
339  }
340  virtual void Free( void* mem ) {
341  if ( !mem ) {
342  return;
343  }
344  --_currentAllocs;
345  Chunk* chunk = (Chunk*)mem;
346 #ifdef DEBUG
347  memset( chunk, 0xfe, sizeof(Chunk) );
348 #endif
349  chunk->next = _root;
350  _root = chunk;
351  }
352  void Trace( const char* name ) {
353  printf( "Mempool %s watermark=%d [%dk] current=%d size=%d nAlloc=%d blocks=%d\n",
354  name, _maxAllocs, _maxAllocs*SIZE/1024, _currentAllocs, SIZE, _nAllocs, _blockPtrs.Size() );
355  }
356 
357  void SetTracked() {
358  _nUntracked--;
359  }
360 
361  int Untracked() const {
362  return _nUntracked;
363  }
364 
365  enum { COUNT = 1024/SIZE }; // Some compilers do not accept to use COUNT in private part if COUNT is private
366 
367 private:
368  union Chunk {
369  Chunk* next;
370  char mem[SIZE];
371  };
372  struct Block {
373  Chunk chunk[COUNT];
374  };
375  DynArray< Block*, 10 > _blockPtrs;
376  Chunk* _root;
377 
378  int _currentAllocs;
379  int _nAllocs;
380  int _maxAllocs;
381  int _nUntracked;
382 };

471  inline static bool StringEqual( const char* p, const char* q, int nChar=INT_MAX ) {
472  int n = 0;
473  if ( p == q ) {
474  return true;
475  }
476  while( *p && *q && *p == *q && n<nChar ) {
477  ++p;
478  ++q;
479  ++n;

482  return true;

484  return false;
