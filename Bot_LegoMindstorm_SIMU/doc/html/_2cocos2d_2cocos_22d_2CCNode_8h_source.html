945  template<typename _T> inline
946  static void sortNodes(cocos2d::Vector<_T*>& nodes)
947  {
948  static_assert(std::is_base_of<Node, _T>::value, "Node::sortNodes: Only accept derived of Node!");
949 #if CC_64BITS
950  std::sort(std::begin(nodes), std::end(nodes), [](_T* n1, _T* n2) {
951  return (n1->_localZOrder$Arrival < n2->_localZOrder$Arrival);
952  });
953 #else
954  std::sort(std::begin(nodes), std::end(nodes), [](_T* n1, _T* n2) {
955  return (n1->_localZOrder == n2->_localZOrder && n1->_orderOfArrival < n2->_orderOfArrival) || n1->_localZOrder < n2->_localZOrder;
956  });
957 #endif
958  }
