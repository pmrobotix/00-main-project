26 /* Copy buffers with conversion between buffer sample formats.
27  *
28  * dst Destination buffer
29  * dst_format Destination buffer format
30  * src Source buffer
31  * src_format Source buffer format
32  * count Number of samples to copy
33  *
34  * Allowed format conversions are given by either case 1 or 2 below:
35  *
36  * 1) One of src_format or dst_format is AUDIO_FORMAT_PCM_16_BIT or
37  * AUDIO_FORMAT_PCM_FLOAT, and the other format type is one of:
38  *
39  * AUDIO_FORMAT_PCM_16_BIT
40  * AUDIO_FORMAT_PCM_FLOAT
41  * AUDIO_FORMAT_PCM_8_BIT
42  * AUDIO_FORMAT_PCM_24_BIT_PACKED
43  * AUDIO_FORMAT_PCM_32_BIT
44  * AUDIO_FORMAT_PCM_8_24_BIT
45  *
46  * 2) Both dst_format and src_format are identical and of the list given
47  * in (1). This is a straight copy.
48  *
49  * The destination and source buffers must be completely separate if the destination
50  * format size is larger than the source format size. These routines call functions
51  * in primitives.h, so descriptions of detailed behavior can be reviewed there.
52  *
53  * Logs a fatal error if dst or src format is not allowed by the conversion rules above.
54  */
55 void memcpy_by_audio_format(void *dst, audio_format_t dst_format,
56  const void *src, audio_format_t src_format, size_t count);
