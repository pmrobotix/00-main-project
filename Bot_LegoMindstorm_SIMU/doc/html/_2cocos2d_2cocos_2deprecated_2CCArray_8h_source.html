52 template < class T >
53 class RCPtr
54 {
55 public:
56  //Construct using a C pointer
57  //e.g. RCPtr< T > x = new (std::nothrow) T();
58  RCPtr(T* ptr = nullptr)
59  : _ptr(ptr)
60  {
61  if(ptr != nullptr) {ptr->retain();}
62  }
63 
64  //Copy constructor
65  RCPtr(const RCPtr &ptr)
66  : _ptr(ptr._ptr)
67  {
68 // printf("Array: copy constructor: %p\n", this);
69  if(_ptr != NULL) {_ptr->retain();}
70  }
71 
72  //Move constructor
73  RCPtr(RCPtr &&ptr)
74  : _ptr(ptr._ptr)
75  {
76 // printf("Array: Move Constructor: %p\n", this);
77  ptr._ptr = nullptr;
78  }
79 
80  ~RCPtr()
81  {
82 // printf("Array: Destructor: %p\n", this);
83  if(_ptr != nullptr) {_ptr->release();}
84  }
85 
86  //Assign a pointer
87  //e.g. x = new (std::nothrow) T();
88  RCPtr &operator=(T* ptr)
89  {
90 // printf("Array: operator= T*: %p\n", this);
91 
92  //The following grab and release operations have to be performed
93  //in that order to handle the case where ptr == _ptr
94  //(See comment below by David Garlisch)
95  if(ptr != nullptr) {ptr->retain();}
96  if(_ptr != nullptr) {_ptr->release();}
97  _ptr = ptr;
98  return (*this);
99  }
100 
101  //Assign another RCPtr
102  RCPtr &operator=(const RCPtr &ptr)
103  {
104 // printf("Array: operator= const&: %p\n", this);
105  return (*this) = ptr._ptr;
106  }
107 
108  //Retrieve actual pointer
109  T* get() const
110  {
111  return _ptr;
112  }
113 
114  //Some overloaded operators to facilitate dealing with an RCPtr
115  //as a conventional C pointer.
116  //Without these operators, one can still use the less transparent
117  //get() method to access the pointer.
118  T* operator->() const {return _ptr;} //x->member
119  T &operator*() const {return *_ptr;} //*x, (*x).member
120  explicit operator T*() const {return _ptr;} //T* y = x;
121  explicit operator bool() const {return _ptr != nullptr;} //if(x) {/*x is not NULL*/}
122  bool operator==(const RCPtr &ptr) {return _ptr == ptr._ptr;}
123  bool operator==(const T *ptr) {return _ptr == ptr;}
124 
125 private:
126  T *_ptr; //Actual pointer
127 };
