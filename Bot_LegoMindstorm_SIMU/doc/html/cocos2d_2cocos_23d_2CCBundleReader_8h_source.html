146 template<typename T>
147 inline bool BundleReader::readArray(unsigned int *length, std::vector<T> *values)
148 {
149  if (!read(length))
150  {
151  return false;
152  }
153 
154  if (*length > 0 && values)
155  {
156  values->resize(*length);
157  if (read(&(*values)[0], sizeof(T), *length) != *length)
158  {
159  return false;
160  }
161  }
162  return true;
163 }
