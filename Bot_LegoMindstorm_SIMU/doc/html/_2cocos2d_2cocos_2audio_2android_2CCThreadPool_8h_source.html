164  template<typename T>
165  class ThreadSafeQueue
166  {
167  public:
168  bool push(T const& value)
169  {
170  std::unique_lock<std::mutex> lock(this->mutex);
171  this->q.push(value);
172  return true;
173  }
174 
175  // deletes the retrieved element, do not use for non integral types
176  bool pop(T& v)
177  {
178  std::unique_lock<std::mutex> lock(this->mutex);
179  if (this->q.empty())
180  return false;
181  v = this->q.front();
182  this->q.pop();
183  return true;
184  }
185 
186  bool empty() const
187  {
188  auto thiz = const_cast<ThreadSafeQueue*>(this);
189  std::unique_lock<std::mutex> lock(thiz->mutex);
190  return this->q.empty();
191  }
192 
193  size_t size() const
194  {
195  auto thiz = const_cast<ThreadSafeQueue*>(this);
196  std::unique_lock<std::mutex> lock(thiz->mutex);
197  return this->q.size();
198  }
199 
200  private:
201  std::queue<T> q;
202  std::mutex mutex;
203  };
