141 // Helper class that retains the original order of a set of identifiers and
142 // also provides quick lookup.
143 template<typename T> class SymbolTable {
144  public:
145  ~SymbolTable() {
146  for (auto it = vec.begin(); it != vec.end(); ++it) {
147  delete *it;
148  }
149  }
150 
151  bool Add(const std::string &name, T *e) {
152  vec.emplace_back(e);
153  auto it = dict.find(name);
154  if (it != dict.end()) return true;
155  dict[name] = e;
156  return false;
157  }
158 
159  T *Lookup(const std::string &name) const {
160  auto it = dict.find(name);
161  return it == dict.end() ? nullptr : it->second;
162  }
163 
164  private:
165  std::map<std::string, T *> dict; // quick lookup
166 
167  public:
168  std::vector<T *> vec; // Used to iterate in order of insertion
169 };
