54  {
55  if (!billboardChain)
56  return;
57 
58  // Set width to 0 if not visible
59  width = visible ? width : 0;
60 
61  size_t max = billboardChain->getMaxChainElements();

63  for (size_t j = 0; j < max; j++)
64  {
65  element = billboardChain->getChainElement(chainIndex, j);
66  element.width = width;
67  billboardChain->updateChainElement(chainIndex, j, element);
