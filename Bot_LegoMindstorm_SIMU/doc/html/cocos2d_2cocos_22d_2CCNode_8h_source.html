934  template<typename _T> inline
935  static void sortNodes(cocos2d::Vector<_T*>& nodes)
936  {
937  static_assert(std::is_base_of<Node, _T>::value, "Node::sortNodes: Only accept derived of Node!");
938 #if CC_64BITS
939  std::sort(std::begin(nodes), std::end(nodes), [](_T* n1, _T* n2) {
940  return (n1->_localZOrderAndArrival < n2->_localZOrderAndArrival);
941  });
942 #else
943  std::stable_sort(std::begin(nodes), std::end(nodes), [](_T* n1, _T* n2) {
944  return n1->_localZOrder < n2->_localZOrder;
945  });
946 #endif
947  }
