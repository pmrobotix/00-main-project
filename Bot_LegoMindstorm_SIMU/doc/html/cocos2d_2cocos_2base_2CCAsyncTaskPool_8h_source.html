104  // thread tasks internally used
105  class ThreadTasks {
106  struct AsyncTaskCallBack
107  {
108  TaskCallBack callback;
109  void* callbackParam;
110  };
111  public:
112  ThreadTasks()
113  : _stop(false)
114  {
115  _thread = std::thread(
116  [this]
117  {
118  for(;;)
119  {
120  std::function<void()> task;
121  AsyncTaskCallBack callback;
122  {
123  std::unique_lock<std::mutex> lock(this->_queueMutex);
124  this->_condition.wait(lock,
125  [this]{ return this->_stop || !this->_tasks.empty(); });
126  if(this->_stop && this->_tasks.empty())
127  return;
128  task = std::move(this->_tasks.front());
129  callback = std::move(this->_taskCallBacks.front());
130  this->_tasks.pop();
131  this->_taskCallBacks.pop();
132  }
133 
134  task();
135  Director::getInstance()->getScheduler()->performFunctionInCocosThread([&, callback]{ callback.callback(callback.callbackParam); });
136  }
137  }
138  );
139  }
140  ~ThreadTasks()
141  {
142  {
143  std::unique_lock<std::mutex> lock(_queueMutex);
144  _stop = true;
145 
146  while(_tasks.size())
147  _tasks.pop();
148  while (_taskCallBacks.size())
149  _taskCallBacks.pop();
150  }
151  _condition.notify_all();
152  _thread.join();
153  }
154  void clear()
155  {
156  std::unique_lock<std::mutex> lock(_queueMutex);
157  while(_tasks.size())
158  _tasks.pop();
159  while (_taskCallBacks.size())
160  _taskCallBacks.pop();
161  }
162  template<class F>
163  void enqueue(const TaskCallBack& callback, void* callbackParam, F&& f)
164  {
165  auto task = f;//std::bind(std::forward<F>(f), std::forward<Args>(args)...);
166 
167  {
168  std::unique_lock<std::mutex> lock(_queueMutex);
169 
170  // don't allow enqueueing after stopping the pool
171  if(_stop)
172  {
173  CC_ASSERT(0 && "already stop");
174  return;
175  }
176 
177  AsyncTaskCallBack taskCallBack;
178  taskCallBack.callback = callback;
179  taskCallBack.callbackParam = callbackParam;
180  _tasks.emplace([task](){ task(); });
181  _taskCallBacks.emplace(taskCallBack);
182  }
183  _condition.notify_one();
184  }
185  private:
186 
187  // need to keep track of thread so we can join them
188  std::thread _thread;
189  // the task queue
190  std::queue< std::function<void()> > _tasks;
191  std::queue<AsyncTaskCallBack> _taskCallBacks;
192 
193  // synchronization
